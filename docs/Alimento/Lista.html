def each
	tmp = @head
	while tmp != nil do
		yield tmp.valor
		tmp = tmp.siguiente
	end
end

def insert(nodo)
	if(nodo.is_a?Nodo) then
		aux = @head
		@head = nodo
		@head.anterior = nil
		@head.siguiente = aux
		aux.anterior = @head
	end
end
